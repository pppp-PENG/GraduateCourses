        \notag \max_{a \in \mathcal{A}} \sum_{s' \in \mathcal{S}} & P(s'|s, a) \left( R(s, a, s') + \gamma V^*(s') \right)
    \end{align}

    \begin{align}
        \notag R = (I - \gamma P) \Phi w
    \end{align}
